START

INPUT n

result = 1

IF n < 0 THEN
    PRINT "Please enter a non-negative integer."
    
END IF

IF n == 0 OR n == 1THEN
    PRINT "The factorial of n is 1."
    
END IF

FOR current_num = 1 TO n
    result = result * current_num 
END FOR

OUTPUT "The factorial of n is :result"

END

//  3!=> result =1 ,current=1 --> 1*1=1
 result =1,current=2 ---> 1*2 =2
 result =2,current=3 ---> 2*3 =6
